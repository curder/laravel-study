# 测试Repository查询

  • 建立文件并编写测试代码，文件在tests/Feature/ArticleRepositoryTest.php
<?php

namespace Tests\Feature;

use App\Article;
use App\Repositories\ArticleRepository;
use Tests\TestCase;
use Illuminate\Foundation\Testing\DatabaseMigrations;

class ArticleRepositoryTest extends TestCase
{
    use DatabaseMigrations;

    /**
     * @var ArticleRepository
     */
    protected $repository;

    protected function setUp()
    {
        parent::setUp();
        // 填充数据
        $this->repository = new ArticleRepository();
    }
    
    // 测试获取最后的10条资料
    /** @test */
    public function fetch_latest_10_articles()
    {
        // given data.
        for ($i = 1; $i <= 100; $i++) {
            factory(Article::class)->create([
                'name' => 'name ' . $i,
                'body' => 'body ' . $i,
            ]);
        }

        // 从 repository 中取得最新 10 条数据
        $articles = $this->repository->latest(10)->get();
        $this->assertCount(10, $articles);

        // 确认标题是从 100 .. 91 倒数
        // "title 100" .. "title 91"
        $i = 10;
        /** @var \App\Article[] $articles */
        foreach ($articles as $article) {
            $this->assertEquals('name ' . $i, $article->name);
            --$i;
        }
    }
}
