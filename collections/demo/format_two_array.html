$lastYear = [
  6345.75,
  9839.45,
  7134.6,
  9479.5,
  9928.0,
  8652.0,
  7658.4,
  10245.4,
  7889.4,
  3892.4,
  3638.4,
  2339.4,
];
$thisYear = [
  6145.75,
  6895.0,
  3434.0,
  9349350,
  9478.6,
  7652.8,
  4758.4,
  10945.4,
  3689.4,
  8992.4,
  7588.4,
  2239.4,
];

$profit = collect($thisYear)
  ->zip($lastYear)
  ->map(function ($monthly) {
    return $monthly->first() - $monthly->last();
  });
dd($profit);
