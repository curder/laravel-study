# 根据日期和供应商分组

$result = collect($orders)->groupBy('date')->map(function($item){
    return $item->groupBy('supplier');
});

# 计算每日销售金额

$result = collect($orders)->groupBy('date')->map(function($item){
    return $item->sum(function($item){
        return $item['item_unit_price'] * $item['qty'];
    });
});
